import mx.controls.Alert;
			import flash.geom.*;
			import flash.display.*;
			import flash.net.*;
			
			private var g:Graphics;
			private var loader:URLLoader;
			private function init():void
			{
				loader=new URLLoader();
				loader.addEventListener(Event.COMPLETE,onComplete);
				loader.load(new URLRequest("/test/2.aspx"));
			}
			
			private function onComplete(e:Event):void
			{
				var xml:XMLList=new XMLList(loader.data).child("S");
				var p:Point=new Point();
				
				//myCanvas.graphics.lineStyle(1,0xff0000);
				myCanvas.graphics.beginFill(0x000000,1);
				for(var i:int=0;i<xml.length();i++)
				{
					if(xml[i].attribute("t")=="0")
					{
						p.x=xml[i].attribute("p").toString().split(",")[0];
						p.y=xml[i].attribute("p").toString().split(",")[1];
						myCanvas.graphics.moveTo(p.x,p.y);
					}
					else if(xml[i].attribute("t")=="1")
					{
						p.x=xml[i].attribute("p").toString().split(",")[0];
						p.y=xml[i].attribute("p").toString().split(",")[1];
						myCanvas.graphics.lineTo(p.x,p.y);
					}
					else if(xml[i].attribute("t")=="2")
					{
						drawCurve(xml[i].attribute("a"),xml[i].attribute("b"),xml[i].attribute("c"),xml[i].attribute("d"));
					}
				}
				myCanvas.graphics.endFill();
			}
			
			private function drawCurve(a:String,b:String,c:String,d:String):void
			{
				var p1:Point=new Point();
				var c1:Point=new Point();
				var c2:Point=new Point();
				var p2:Point=new Point();
				p1.x=a.split(",")[0];
				p1.y=a.split(",")[1];
				c1.x=b.split(",")[0];
				c1.y=b.split(",")[1];
				c2.x=c.split(",")[0];
				c2.y=c.split(",")[1];
				p2.x=d.split(",")[0];
				p2.y=d.split(",")[1];
				
				var anchor1:Point, anchor2:Point, anchor3:Point, anchor4:Point, anchor5:Point;
	            var control1:Point, control2:Point, control3:Point, control4:Point;
				var m1:Point,m2:Point,m3:Point,m4:Point,m5:Point,m6:Point,m7:Point,m8:Point,m9:Point;
	            anchor1 = p1;
	            anchor5 = p2;
	
	            m1 = GetMidPoint(p1, c1);
	            m2 = GetMidPoint(c1, c2);
	            m3 = GetMidPoint(c2, p2);
	            m4 = GetMidPoint(m1, m2);
	            m5 = GetMidPoint(m2, m3);
	
	            anchor3 = GetMidPoint(m4, m5);
	
	            m6 = GetMidPoint(p1, m1);
	            m7 = GetMidPoint(p2, m3);
	
	            m8 = GetMidPoint(m4, anchor3);
	            m9 = GetMidPoint(anchor3, m5);
	
	            control1 = GetMidPoint(m6, m1);
	            control2 = GetMidPoint(m4, m8);
	            control3 = GetMidPoint(m9, m5);
	            control4 = GetMidPoint(m3, m7);
	
	            anchor2 = GetMidPoint(control1, control2);
	            anchor4 = GetMidPoint(control3, control4);
            
            	myCanvas.graphics.curveTo(control1.x,control1.y,anchor2.x,anchor2.y);
            	myCanvas.graphics.curveTo(control2.x,control2.y,anchor3.x,anchor3.y);
            	myCanvas.graphics.curveTo(control3.x,control3.y,anchor4.x,anchor4.y);
            	myCanvas.graphics.curveTo(control4.x,control4.y,anchor5.x,anchor5.y);
			}
			
			private function GetMidPoint(s:Point,e:Point):Point
			{
				return Point.interpolate(s,e,0.5);
			}
